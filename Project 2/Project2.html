from sklearn.ensemble import RandomForestClassifier
randForest = RandomForestClassifier()
randForest.fit(X_train, y_train)

# When we're scoring our method, we need to know a few things:
# Accuracy: the measure of the overall correctness of our model's predictions
# Precision: how accurate the model is when predicting our positive class which is our hazardous objects
# Recall: how well our model does at identifying the positive instances within our dataset - to me this score is the most important, because we'd much
# rather misclassify non-hazardous objects as hazardous, instead of hazardous objects as non-hazardous.
# F1: A harmonic mean of our model's precision and recall scores 

print(f"Score: {randForest.score(X_test, y_test) * 100:.2f}%")
# This gives us the accuracy score, which is at around 92% currently. This doesn't give us a lot of information though, so we should break it down
# further by getting the Precision, Recall, and F1 scores.

y_predict = randForest.predict(X_test)
accuracy = accuracy_score(y_test, y_predict)
print(f"Accuracy: {accuracy * 100:.2f}%")
precision = precision_score(y_test, y_predict)
print(f"Precision: {precision * 100:.2f}%")
recall = recall_score(y_test, y_predict)
print(f"Recall: {recall * 100:.2f}%")
f1 = f1_score(y_test, y_predict)
print(f"F1: {f1 * 100:.2f}%")

# from our initial scores above, we can see that our model does really well with predicting negative values - our non-hazardous NEO's,
# but is lacking when we're predicting our hazardous objects. This can be due to a few things: The high amount of overlap between our two objects,
# the fact that our data set is pretty unbalanced with a ratio of about 90% of our entries being non-hazardous, and 10% being hazardous. 

# To improve our model we can try to adjust the weights of our predictions, meaning that we can make it more important for our model to correctly predict
# hazardous objects than non-hazardous, as well as adjust the threshold that it makes its predictions at. By sacrificing some of our precision, we
# can improve our recall value, which in my eyes is significantly more important in this context.
# The threshold is a value that indicates the percentage of weights required to classify an object, and the optimal threshold is where the difference
# between the true positive rate and false positive rate is at its lowest. One way we can calculate this is by testing for each value of the threshold
# to see which gives us the best f1 score. The best one I found was at around .3-.35
In [20]:
# to assign weights to our classes, we can take a proportion of the sample size vs (2 * the class sample) to get the appropriate weight for each.
TP = 2509
TN = 28589
FP = 992
FN = 1728
total_predictions = TP+TN+FP+FN # this gives us the amount of values in our testing set (looking back, I couldve just taken the total for the test size
# from the shape I took earlier.
total_false = TN+FP # this gives us the number of non-hazardous objects in our testing set
total_true = TP+FN # this gives us the number of hazardous objects in our testing set
weight1 = total_predictions/(2*total_false) # these are our new weights according to our formula
weight2 = total_predictions/(2*total_true) 
class_weights = {0: weight1, 1: weight2}
randForest = RandomForestClassifier(class_weight = class_weights, random_state=42)
randForest.fit(X_train, y_train)

# In order to use a different threshold, we should use the predict_proba method instead. This computes the predicted class probabilities of an input
# sample as the mean of thhe predicted class probabilities of the trees in the forest.
y_predict = randForest.predict_proba(X_test)

# This is our new threshold!
new_threshold = .3

# We take the array generated above, selecting all of the positive probability predictions, and compare them to our new probability threshold. 
# Then our boolean result is casted to either a 0 or a 1 and used to compare and calculate the scores for our model.
# : selects all of the rows, and the 1 value selects the probability of the positive class in the generated array to compare to our new threshold.
y_predict_threshold = (y_predict[:,1] >= new_threshold).astype(int)
accuracy = accuracy_score(y_test, y_predict_threshold)
print(f"Accuracy: {accuracy * 100:.2f}%")
precision = precision_score(y_test, y_predict_threshold)
print(f"Precision: {precision * 100:.2f}%")
recall = recall_score(y_test, y_predict_threshold)
print(f"Recall: {recall * 100:.2f}%")
f1 = f1_score(y_test, y_predict_threshold)
print(f"F1: {f1 * 100:.2f}%")
print(confusion_matrix(y_test, y_predict_threshold))
